class Client(db.Model): id = db.Column(db.Integer, primary_key=True) name = db.Column(db.String(100), nullable=False) age = db.Column(db.Integer, nullable=False) requests = db.relationship('Request', backref='client', lazy=True) @app.route('/add_client', methods=['GET', 'POST']) def add_client(): if request.method == 'POST': name = request.form['name'] age = request.form['age'] new_client = Client(name=name, age=int(age)) db.session.add(new_client) db.session.commit() return redirect(url_for('index')) return render_template('add_client.html')
